create table 'user' (
	id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
	username varchar(50),
	password varchar(50),
	name varchar(50),
	card_id varchar(50),
	
	phone_number varchar(50),
	email varchar(50),
	address varchar(100),
	package_id varchar(50),
	dealer_id int

)
